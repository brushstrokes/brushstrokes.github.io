What is the runtime of each method? Give your answer in Θ notation as a function of n. 
int foo1(int n) {
   int x = 0;
   for (int i = 0; i < n; ++i)
      for (int j = 0; j < i; ++j)
         ++x;
   return x;
}

int foo2(int n) {
   int x = 0;
   while (n > 0) {
      n = n / 2;
      ++x;
   }
   return x;
}
